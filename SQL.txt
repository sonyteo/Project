## SQL Codes ##

CREATE TABLE Users (
user_id INT(6) NOT NULL AUTO_INCREMENT,
username VARCHAR(30) NOT NULL,
password VARCHAR(255) NOT NULL,
email VARCHAR(50) NOT NULL,
reg_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
isAdmin TINYINT(1) NOT NULL DEFAULT 0,
PRIMARY KEY (user_id)
);

CREATE TABLE Hotels (
hotel_id INT(6) NOT NULL AUTO_INCREMENT,
hotel_name VARCHAR(50) NOT NULL,
hotel_img VARCHAR(50) NOT NULL,
location_id INT(6) NOT NULL,
price INT(11) NOT NULL,
description VARCHAR(255) NOT NULL,
PRIMARY KEY (hotel_id)
);

CREATE TABLE Rooms (
room_id INT(6) NOT NULL AUTO_INCREMENT,
room_type VARCHAR(50) NOT NULL,
room_img VARCHAR(255) NOT NULL,
price INT(6) NOT NULL,
max_person INT(6) NOT NULL,
description VARCHAR(255),
PRIMARY KEY (room_id)
);

CREATE TABLE Booking (
booking_id INT(6) NOT NULL,
user_id VARCHAR(255) NOT NULL,
hotel_id INT(6) NOT NULL,
room_id INT(6) NOT NULL,
bedding VARCHAR(255),
noOfRoom TINYINT(2) NOT NULL,
meal VARCHAR(255),
cin_date DATE NOT NULL,
cout_date DATE NOT NULL,
remarks VARCHAR(255),
book_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
payment TINYINT(1) NOT NULL DEFAULT 0,
PRIMARY KEY (booking_id)
);

CREATE TABLE subscribe (
sub_id INT(6) NOT NULL,
sub_email VARCHAR(50),
sub_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (sub_id)
);

CREATE TABLE location (
location_id INT(6) NOT NULL AUTO_INCREMENT,
location VARCHAR(50) NOT NULL,
location_img VARCHAR(50) NOT NULL,
PRIMARY KEY (location_id)
);

CREATE TABLE rating (
rate_id INT(6) NOT NULL,
user_id VARCHAR(255), 
hotel_id INT(6) NOT NULL,
rate INT(6) NOT NULL,
review_title VARCHAR(50) NOT NULL,
review VARCHAR(255),
rate_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (rate_id)
);
